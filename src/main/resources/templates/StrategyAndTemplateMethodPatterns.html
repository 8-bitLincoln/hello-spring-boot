* Pattern TEMPLATE METHOD:

class Parent {
    protected firstStep();
    protected secondStep();
    public templateMethod(){
        firstStep();
        secondStep();
    }
}

class Child extend Parent {
    protected firstStep() {...};
    protected secondStep() {...};
}

public static void main(String[] args) {
child.templateMethod();
}

* Pattern STRATEGY:

interface Strategy {
    execute();
}

class StrategyOne implements Strategy {
    execute() {...};
}

class StrategyTwo implements Strategy {
    execute() {...};
}

class Context {
    private Strategy;

    setStrategy(Strategy strategy) {...};
    executeStrategy() { strategy.execute() }
}
